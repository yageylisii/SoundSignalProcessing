cmake_minimum_required(VERSION 3.16)

project(sound_processor LANGUAGES CXX)

option(BUILD_TESTS "Build tests" OFF)

set(SOUND_PROCESSOR_HEADERS
    include/application.hpp
    include/parser/console_parser.hpp
    include/pipeline/cmd_line_args_2_pipeline_converter.hpp
    include/pipeline/pipeline.hpp
    include/filters/common/filter.hpp
    include/filters/filters.hpp
    include/filters/generator/am_generator_filter.hpp
    include/filters/generator/fm_generator_filter.hpp
    include/filters/generator/sin_generator_filter.hpp
    include/filters/transform/ampl_filter.hpp
    include/filters/transform/low_pass_filter.hpp
    include/filters/transform/normalize_filter.hpp
    include/filters/transform/silence_filter.hpp
    include/filters/transform/time_stretch_filter.hpp
    include/wav/wav_format.hpp
    include/wav/wav_reader.hpp
    include/wav/wav_writer.hpp
    include/wav/waveform.hpp
)

set(SOUND_PROCESSOR_SOURCES
    src/application.cpp
    src/parser/console_parser.cpp
    src/pipeline/cmd_line_args_2_pipeline_converter.cpp
    src/filters/common/filter_utils.cpp
    src/filters/generator/am_generator_filter.cpp
    src/filters/generator/fm_generator_filter.cpp
    src/filters/generator/sin_generator_filter.cpp
    src/filters/transform/ampl_filter.cpp
    src/filters/transform/low_pass_filter.cpp
    src/filters/transform/normalize_filter.cpp
    src/filters/transform/silence_filter.cpp
    src/filters/transform/time_stretch_filter.cpp
    src/pipeline/pipeline.cpp
    src/wav/waveform.cpp
    src/wav/wav_reader.cpp
    src/wav/wav_writer.cpp
)

add_executable(
    sound_processor
    app/main.cpp
    ${SOUND_PROCESSOR_HEADERS}
    ${SOUND_PROCESSOR_SOURCES}
)
target_include_directories(sound_processor PRIVATE include)
target_compile_features(sound_processor PRIVATE cxx_std_17)

if(BUILD_TESTS)
    find_package(Catch2 3 QUIET)
    if(NOT Catch2_FOUND)
        include(FetchContent)
        FetchContent_Declare(
            Catch2
            GIT_REPOSITORY https://github.com/catchorg/Catch2.git
            GIT_TAG v3.4.0
        )
        FetchContent_MakeAvailable(Catch2)
    endif()

    enable_testing()

    add_executable(
        sound_processor_tests
        tests/sound_processor_tests.cpp
        ${SOUND_PROCESSOR_HEADERS}
        ${SOUND_PROCESSOR_SOURCES}
    )
    target_include_directories(sound_processor_tests PRIVATE include)
    target_compile_features(sound_processor_tests PRIVATE cxx_std_17)
    target_link_libraries(
        sound_processor_tests
        PRIVATE Catch2::Catch2WithMain
    )

    add_test(
        NAME SoundProcessorTests
        COMMAND $<TARGET_FILE:sound_processor_tests>
    )
endif()
